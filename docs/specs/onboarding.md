function calculateProfileCompletion(state: ProfileCompletionState): number {
  const phase1Weight = 40 // 40% do perfil
  const phase2Weight = 60 // 60% do perfil
  
  let score = 0
  
  // Phase 1
  if (state.phase1.completed) {
    score += phase1Weight
  }
  
  // Phase 2 (cada step vale 12%)
  const phase2Steps = Object.values(state.phase2.steps)
  const completedSteps = phase2Steps.filter(s => s.completed).length
  score += (completedSteps / phase2Steps.length) * phase2Weight
